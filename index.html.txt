<script>
  const servicos = ['limpeza','fundacao','estrutura','eletrica','acabamento'];
  servicos.forEach(s => {
    document.getElementById(s).addEventListener('change', e => {
      document.getElementById(s + '_mat').disabled = !e.target.checked;
      document.getElementById(s + '_mao').disabled = !e.target.checked;
      if (!e.target.checked) {
        document.getElementById(s + '_mat').value = '';
        document.getElementById(s + '_mao').value = '';
      }
    });
  });

  function calcularOrcamento() {
    const local = document.getElementById('local').value.trim();
    const tipo = document.getElementById('tipo').value.trim();
    const area = parseFloat(document.getElementById('area').value);

    if (!local || !tipo || isNaN(area) || area <= 0) {
      alert('Preencha corretamente Localização, Tipo de obra e Área.');
      return;
    }

    let totalMaterial = 0;
    let totalMao = 0;
    let servicosSelecionados = [];

    for (let s of servicos) {
      if (document.getElementById(s).checked) {
        let mat = parseFloat(document.getElementById(s + '_mat').value);
        let mao = parseFloat(document.getElementById(s + '_mao').value);

        if (isNaN(mat) || mat < 0 || isNaN(mao) || mao < 0) {
          alert('Informe valores válidos para materiais e mão de obra em ' + s.charAt(0).toUpperCase() + s.slice(1));
          return;
        }
        servicosSelecionados.push(s.charAt(0).toUpperCase() + s.slice(1));
        totalMaterial += mat;
        totalMao += mao;
      }
    }

    if (servicosSelecionados.length === 0) {
      alert('Selecione ao menos um serviço.');
      return;
    }

    const total = totalMaterial + totalMao;
    const sinal = total * 0.3;
    const meio = total * 0.4;
    const final = total * 0.3;

    let texto = `Orçamento para obra em ${local}\n`;
    texto += `Tipo de obra: ${tipo}\n`;
    texto += `Área: ${area.toFixed(2)} m²\n\n`;
    texto += `Serviços incluídos:\n`;
    for (let s of servicosSelecionados) texto += `- ${s}\n`;
    texto += `\nCustos:\n- Materiais: R$ ${totalMaterial.toFixed(2)}\n- Mão de obra: R$ ${totalMao.toFixed(2)}\n`;
    texto += `Total: R$ ${total.toFixed(2)}\n\n`;
    texto += `Formas de pagamento:\n- Sinal (30%): R$ ${sinal.toFixed(2)}\n- Durante a obra (40%): R$ ${meio.toFixed(2)}\n- Na entrega (30%): R$ ${final.toFixed(2)}\n`;

    document.getElementById('resultado').textContent = texto;
  }
</script>
